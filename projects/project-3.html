Python - Automated ML Pipeline 
import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestRegressor
from xgboost import XGBRegressor
from lightgbm import LGBMRegressor
from optuna import create_study
import mlflow

class AutoMLPipeline:
    def __init__(self, target_metric='rmse'):
        self.target_metric = target_metric
        self.models = {
            'xgboost': XGBRegressor(),
            'lightgbm': LGBMRegressor(),
            'random_forest': RandomForestRegressor()
        }
        self.best_model = None
        self.best_params = None
        self.feature_importance = None
    
    def feature_engineering(self, df, date_col, target_col):
        """Automated feature engineering for time series"""
        
        # Convert date column
        df[date_col] = pd.to_datetime(df[date_col])
        df = df.sort_values(date_col)
        
        # Time-based features
        df['year'] = df[date_col].dt.year
        df['month'] = df[date_col].dt.month
        df['day'] = df[date_col].dt.day
        df['dayofweek'] = df[date_col].dt.dayofweek
        df['quarter'] = df[date_col].dt.quarter
        
        # Lag features
        for lag in [1, 7, 30, 365]:
            df[f'{target_col}_lag_{lag}'] = df[target_col].shift(lag)
        
        # Rolling statistics
        for window in [7, 30, 90]:
            df[f'{target_col}_rolling_mean_{window}'] = df[target_col].rolling(window).mean()
            df[f'{target_col}_rolling_std_{window}'] = df[target_col].rolling(window).std()
        
        # Trend and seasonality
        df[f'{target_col}_pct_change'] = df[target_col].pct_change()
        df[f'{target_col}_diff'] = df[target_col].diff()
        
        return df.dropna()
    
    def optimize_hyperparameters(self, X_train, y_train, X_val, y_val, model_name):
        """Bayesian hyperparameter optimization"""
        
        def objective(trial):
            if model_name == 'xgboost':
                params = {
                    'n_estimators': trial.suggest_int('n_estimators', 100, 1000),
                    'max_depth': trial.suggest_int('max_depth', 3, 10),
                    'learning_rate': trial.suggest_float('learning_rate', 0.01, 0.3),
                    'subsample': trial.suggest_float('subsample', 0.8, 1.0)
                }
                model = XGBRegressor(**params, random_state=42)
            
            elif model_name == 'lightgbm':
                params = {
                    'n_estimators': trial.suggest_int('n_estimators', 100, 1000),
                    'max_depth': trial.suggest_int('max_depth', 3, 10),
                    'learning_rate': trial.suggest_float('learning_rate', 0.01, 0.3),
                    'feature_fraction': trial.suggest_float('feature_fraction', 0.8, 1.0)
                }
                model = LGBMRegressor(**params, random_state=42)
            
            else:  # random_forest
                params = {
                    'n_estimators': trial.suggest_int('n_estimators', 100, 500),
                    'max_depth': trial.suggest_int('max_depth', 3, 20),
                    'min_samples_split': trial.suggest_int('min_samples_split', 2, 10)
                }
                model = RandomForestRegressor(**params, random_state=42)
            
            model.fit(X_train, y_train)
            predictions = model.predict(X_val)
            rmse = np.sqrt(np.mean((y_val - predictions) ** 2))
            
            return rmse
        
        study = create_study(direction='minimize')
        study.optimize(objective, n_trials=100)
        
        return study.best_params, study.best_value

Python - Model Serving & Monitoring 
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import mlflow.pyfunc
import pandas as pd
import logging
from datetime import datetime

app = FastAPI(title="Predictive Analytics API")

# Load model
model = mlflow.pyfunc.load_model("models:/sales_forecasting/production")

class PredictionRequest(BaseModel):
    features: dict
    series_id: str

class ModelMonitor:
    def __init__(self):
        self.prediction_counts = {}
        self.error_counts = {}
        self.response_times = []
    
    def log_prediction(self, series_id: str, prediction: float, 
                      actual: float = None, response_time: float = None):
        """Log prediction for monitoring"""
        timestamp = datetime.now()
        
        # Update counters
        self.prediction_counts[series_id] = self.prediction_counts.get(series_id, 0) + 1
        
        if response_time:
            self.response_times.append(response_time)
        
        # Log to MLflow
        with mlflow.start_run(run_name=f"prediction_{timestamp}"):
            mlflow.log_metric("prediction_value", prediction)
            if actual:
                error = abs(prediction - actual)
                mlflow.log_metric("prediction_error", error)
            if response_time:
                mlflow.log_metric("response_time_ms", response_time)
    
    def detect_drift(self, recent_errors: list, historical_mean: float):
        """Simple drift detection"""
        if len(recent_errors) < 10:
            return False
        
        recent_mean = np.mean(recent_errors)
        return recent_mean > historical_mean * 1.5  # 50% increase threshold

monitor = ModelMonitor()

@app.post("/predict")
async def predict(request: PredictionRequest):
    """Generate prediction"""
    start_time = datetime.now()
    
    try:
        # Prepare input
        input_df = pd.DataFrame([request.features])
        
        # Generate prediction
        prediction = model.predict(input_df)[0]
        
        # Calculate response time
        response_time = (datetime.now() - start_time).total_seconds() * 1000
        
        # Log for monitoring
        monitor.log_prediction(
            series_id=request.series_id,
            prediction=prediction,
            response_time=response_time
        )
        
        return {
            "prediction": float(prediction),
            "series_id": request.series_id,
            "timestamp": datetime.now().isoformat(),
            "response_time_ms": response_time
        }
    
    except Exception as e:
        monitor.error_counts["prediction_errors"] = monitor.error_counts.get("prediction_errors", 0) + 1
        raise HTTPException(status_code=500, detail=str(e))
